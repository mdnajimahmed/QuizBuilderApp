# The list of configurations

- Compulsory:
    - POOL_ID (Amazon cognito user pool id)
    - DB_PASSWORD
    - DB_USERNAME
    - DB_URL
    - DB_NAME
    - WEB_CLIENT_ID
    - WEB_REDIRECT_URI
    - WEB_CLIENT_SECRET (compulsory to run the tests in local)
- Optional:
    - SERVER_PORT (default 8080)
    - LOGGING_LEVEL(default: DEBUG)
